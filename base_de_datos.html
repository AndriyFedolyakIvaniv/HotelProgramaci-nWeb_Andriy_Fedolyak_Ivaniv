CREATE TABLE persona(
	id_persona INT AUTO_INCREMENT PRIMARY KEY,
	nom varchar (30) NOT NULL,
	cognom varchar (30) NOT NULL,
	adreça varchar (75) NOT NULL,
	document_identitat varchar (9) NOT NULL,
	data_naixament DATE NOT NULL,
	telefon INT (9) NOT NULL,
	email varchar (30) NOT NULL
	);
CREATE TABLE client(
	id_client INT AUTO_INCREMENT PRIMARY KEY,
	data_registre DATE NOT NULL,
	tipus_client varchar (15) NOT NULL,
	targeta_client INT (12) NOT NULL,
	id_persona INT NOT NULL,
	FOREIGN KEY (id_persona) REFERENCES persona (id_persona)
	);
CREATE TABLE empleat(
	id_empleat INT AUTO_INCREMENT PRIMARY KEY,
	llocFeina varchar (25) NOT NULL,
	data_contratacio DATE NOT NULL,
	salariBrut DECIMAL (10, 2) NOT NULL,
	estat_laboral varchar (15) NOT NULL,
	id_persona INT NOT NULL,
	FOREIGN KEY (id_persona) REFERENCES persona (id_persona)
	);
CREATE TABLE reserva(
	id_reserva INT AUTO_INCREMENT PRIMARY KEY,
	data_reserva DATE NOT NULL,
	data_inici DATE NOT NULL,
	data_fi DATE NOT NULL,
	tipus_reserva varchar (2) NOT NULL,
	tipus_IVA DECIMAL (10, 2) NOT NULL,
	preu_total_reserva DECIMAL (10, 2) NOT NULL,
	id_client INT NOT NULL,
	FOREIGN KEY (id_client) REFERENCES client (id_client),
	id_habitacio INT NOT NULL,
	FOREIGN KEY (id_habitacio) REFERENCES habitacio (id_habitacio)
	);
CREATE TABLE habitacio(
	id_habitacio INT AUTO_INCREMENT PRIMARY KEY,
	numero_habitacio INT NOT NULL,
	tipus varchar (15) NOT NULL,
	capacitat INT NOT NULL,
	preu_nit_AD DECIMAL (10, 2) NOT NULL,
	preu_nit_MP DECIMAL (10, 2) NOT NULL,
	estat varchar (15) NOT NULL,
	descripcio varchar (85) NOT NULL
	);
CREATE TABLE factura(
	id_factura INT AUTO_INCREMENT PRIMARY KEY,
	data_emisio DATE NOT NULL,
	metode_pagament varchar (25) NOT NULL,
	base_imposable DECIMAL (10, 2) NOT NULL,
	IVA DECIMAL (10, 2) NOT NULL,
	total DECIMAL (10, 2) NOT NULL,
	id_reserva INT NOT NULL,
	FOREIGN KEY(id_reserva) REFERENCES reserva(id_reserva)
	);
CREATE TABLE tasca(
	id_tasca INT AUTO_INCREMENT PRIMARY KEY,
	descripcio varchar (25) NOT NULL,
	data_creacio DATE NOT NULL,
	data_ejecucio DATE NOT NULL,
	estat varchar (25) NOT NULL
	);
CREATE TABLE realitzar(
	id_tasca INT(11),
	id_empleat INT(11),
	PRIMARY KEY(id_tasca, id_empleat),
	FOREIGN KEY(id_tasca) REFERENCES tasca (id_tasca),
	FOREIGN KEY(id_empleat) REFERENCES empleat (id_empleat)
	);
